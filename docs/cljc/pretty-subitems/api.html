Pretty Subitems
---
Subitems are additional items within a UI component.
E.g., A menu bar component within a dropdown menu component.
usage
[my-component :my-component
              {:label      "My component"
               :my-subitem {:label "My subitem"}}]

Pretty Subitem Groups
---
Subitem groups are group of subitems within a UI component.
E.g., A group of menu items within a menu bar component.
usage
[my-component :my-component
              {:label              "My component"
               :my-subitem-default {:fill-color :default}
               :my-subitems        [{:label "My subitem #1"} {:label "My subitem #2"}]}]
---
Default property maps of subitem groups are applied on each group item.

Keys of default property maps of subitem groups are derived from the keys of subitem groups.
E.g., ':menu-items' => ':menu-item-default'

(defn apply-group-item-prototype
  [props subitem-group-key prototype-f]
  (letfn [(f0 [group]     (vector/->items group f1 {:provide-dex? true}))
          (f1 [dex props] (prototype-f dex props))]
         (-> props (update subitem-group-key f0))))

(defn subitem-group<-disabled-state
  [{:keys [disabled?] :as props} subitem-group-key]
  (if (-> disabled? nil?)
      (-> props)
      (-> props (update subitem-group-key vector/update-all-item assoc :disabled? disabled?))))

(defn subitem-group<-subitem-default
  [props subitem-group-key]
  (let [subitem-default-key (subitem-default-key subitem-group-key)]
       (if-let [subitem-default (get props subitem-default-key)]
               (-> props (update subitem-group-key vector/update-all-item map/reversed-deep-merge subitem-default))
               (-> props))))

(defn subitem<-disabled-state
  [{:keys [disabled?] :as props} subitem-key]
  (cond (-> disabled? nil?)         (-> props)
        (-> props subitem-key nil?) (-> props)
        :else (-> props (assoc-in [subitem-key :disabled?] disabled?))))
