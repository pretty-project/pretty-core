background-color-rules
(defn background-color-rules
  [{:keys [disabled? highlighted?] :as props}]
  (cond-> props (-> disabled?)        (dissoc :hover-color     :hover-pattern)
                (-> highlighted? not) (dissoc :highlight-color :highlight-pattern)))
description
- Dissociates the ':hover-color' and ':hover-pattern' properties (from the given property map) of disabled elements.
- Dissociates the ':highlight-color' and ':highlight-pattern' properties (from the given property map) of non-highlighted elements.
usage
(background-color-rules {:disabled? true :fill-color :muted :hover-color :muted ...})
=>
{:disabled?  true
 :fill-color :muted
 ...}
usage
(background-color-rules {:highlight-color :muted ...})
=>
{...}
{:disabled? (boolean)(opt)
 :highlighted? (boolean)(opt)
 ...}

border-rules
(defn border-rules
  [{:keys [border-crop border-radius] :as props}]
  (case border-crop :auto (cond-> props border-radius (assoc :border-crop :crop))
                          (->     props)))
description
If the ':border-crop' property is provided as ':auto',
... it is replaced with the value ':crop' in case the ':border-radius' property is also provided.
... it is replaced with NIL if the ':border-radius' property is not provided.
usage
(border-rules {:border-crop :auto :border-radius {:all :s} ...})
=>
{:border-crop   :crop
 :border-radius {:all :s}
 ...}
{:border-crop (keyword)(opt)
 :border-radius (map)(opt)

(defn attached-icon-rules

  (if icon (-> props (map/use-default-values {:icon-color text-color :icon-size font-size}))

for the ':icon-color' and ':icon-size' properties; in case the ':icon' property is provided.

  [{:keys [href-uri on-click-f] :as props}]
  (cond on-click-f (map/use-default-values props {:text-color :default})

mouse-event-rules
(defn mouse-event-rules
  [{:keys [disabled? href-uri on-click-f on-mouse-down-f on-mouse-up-f] :as props}]
  (letfn [(f0 [e] #?(:clj  (do (when on-mouse-up-f (on-mouse-up-f e)))
                     :cljs (do (when on-mouse-up-f (on-mouse-up-f e))
                               (when :always       (dom/blur-active-element!)))))]
         (cond disabled?       (-> props (dissoc :on-click-f :on-mouse-down-f :on-mouse-leave-f :on-mouse-over-f :on-mouse-up-f :on-right-click-f))
               on-click-f      (-> props (map/use-default-values {:on-mouse-up-f f0}))
               on-mouse-down-f (-> props (map/use-default-values {:on-mouse-up-f f0}))
               on-mouse-up-f   (-> props (map/use-default-values {:on-mouse-up-f f0}))
               href-uri        (-> props (map/use-default-values {:on-mouse-up-f f0}))
               :else           (-> props))))
description
- Dissociates the mouse event properties of disabled elements.
- In case the ':href-uri', ':on-click-f', ':on-mouse-down-f' or ':on-mouse-up-f' property is provided,
  it extends the ':on-mouse-up-f' function with the 'dom.api/blur-active-element!' function
  in the given property map to prevent clickable elements staying focused after a click event.
usage
(mouse-event-rules {:disabled? true :on-click-f (fn [_] ...)})
=>
{:disabled? true
 ...}
usage
(mouse-event-rules {:on-click-f (fn [_] ...) ...})
=>
{:on-click-f    (fn [_] ...)
 :on-mouse-up-f (fn [_] ... (dom.api/blur-active-element!))
 ...}
{:disabled? (boolean)(opt)
 :href-uri (string)(opt)
 :on-click-f (function)(opt)
 :on-mouse-down-f (function)(opt)
 :on-mouse-up-f (function)(opt)
 ...}

flex-rules
(defn flex-rules
  [{:keys [horizontal-align orientation overflow vertical-align] :as props}]
  (or (case overflow :scroll (case orientation :horizontal (-> props (assoc :horizontal-align (case horizontal-align :center :left nil :left horizontal-align)))
                                               :vertical   (-> props (assoc :vertical-align   (case vertical-align   :center :top  nil :top  vertical-align)))))
      (-> props)))
description
In case the ':overflow' property is provided as ':scroll' ...
... and the ':orientation' property is provided as ':horizontal' ...
    ... it replaces ':center' or NIL ':horizontal-align' values with ':left'.
... and the ':orientation' property is provided as ':vertical' ...
    ... it replaces ':center' or NIL ':vertical-align' values with ':top'.
usage
(flex-rules {:orientation :horizontal :overflow :scroll ...})
=>
{:horizontal-align :left
 :orientation      :horizontal
 :overflow         :scroll
 ...}
{:horizontal-align (keyword)(opt)
 :orientation (keyword)(opt)
 :overflow (keyword)(opt)
 :vertical-align (keyword)(opt)
